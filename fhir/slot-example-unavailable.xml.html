Slot-example-unavailable.xml

Example of a slot that has been marked as unavailable (id = "3")

Raw XML

<Slot xmlns="http://hl7.org/fhir">
  <text>
    <status value="generated"/>
    <div xmlns="http://www.w3.org/1999/xhtml">
      25 Dec 2013 9:30am - 9:45am: <b>Unavailable</b> Physiotherapy<br/>
      Dr Careful is out of the office
    </div>
  </text>
  <type>
    <coding>
      <code value="45"/>
      <display value="Physiotherapy"/>
    </coding>
  </type>
  <availability>
    <reference value="Availability/example"/>
  </availability>
  <freeBusyType value="BUSY-UNAVAILABLE"/>
  <start value="2013-12-25T09:30:00Z"/>
  <end value="2013-12-25T09:45:00Z"/>
  <comment value="Dr Careful is out of the office"/>
  <author>
    <reference value="Practitioner/example"/>
    <display value="Dr Adam Careful"/>
  </author>
  <authorDate value="2012-09-17"/>
</Slot>
